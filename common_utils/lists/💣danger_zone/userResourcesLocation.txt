var grab = (async function(fileUrl) {
  'use strict';
  let response = await fetch(fileUrl);
  if (response.ok) { // if HTTP-status is 200-299
    let text = await response.text();
    return text
  } else {
    console.log("HTTP-Error: " + response.status);
  }
})();
